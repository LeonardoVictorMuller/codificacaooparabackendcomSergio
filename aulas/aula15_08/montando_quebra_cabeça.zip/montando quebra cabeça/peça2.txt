        const errorPath = path.join(__dirname, 'erro.html');
        fs.readFile(errorPath, (err, data) > {
            if (err) {
                res.writeHead(500, { 'Content-Type': 'text/html' });
                res.end('<h1>Erro interno no servidor</h1>');
                return;
            }

        });
